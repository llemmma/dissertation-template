This chapter develops the main technical apparatus.  Building on
the Kripke semantics of Chapter~\ref{ch:chapter1}, we discuss
frame conditions and introduce neighborhood semantics.

\section{Frame Conditions}

Properties of the accessibility relation correspond to axiom
schemas.

\begin{proposition}\label{prop:reflexivity}
  A frame $\FR$ validates $\tax$ ($\nec A \to A$) if and only if
  $R$ is reflexive.
\end{proposition}

\begin{proof}
  (LTR)\quad Fix $w \in W$ and set $V(p) = \set{v :
  wRv}$.  Let $\mwrv$.  Then $\M,w\trues \nec p$, so $\tax$ gives $\M,w\trues
  p$, hence $w \in V(p)$, i.e., $wRw$.

  (RTL)\quad If $R$ is reflexive and $\M,w\trues \nec A$,
  then $wRw$ gives $\M,w\trues A$.
\end{proof}

\begin{note}\label{note:correspondences}
  Transitivity corresponds to $\nec A \to \nec\nec A$ and
  symmetry to $A \to \nec\poss A$.  See
  \textcite[ch.~3]{blackburn2001}.
\end{note}

\section{Neighborhood Semantics}

Neighborhood semantics generalize relational semantics by
replacing $R$ with a set-selection function
\cite{scott1970, pacuit2017}.

\begin{definition}\label{def:nbhd-model}
  A \textit{neighborhood model} $\MN = \oset{W, \fn, V}$ has $\fn : W \to \power{\power{W}}$ assigning each world
  a collection of neighborhoods.  The modal clause is:
  \[
    \MN,w\trues \nec A \text{ if and only if }
    \truthset{A}^{\MN} \in \fn(w).
  \]
\end{definition}

\begin{remark}\label{rem:generalization}
  Every Kripke frame induces a neighborhood frame via $\fn(w) =
  \set{X \subseteq W : \set{v : wRv} \subseteq X}$.  The converse
  fails: neighborhood frames can model non-normal logics lacking
  $\kax$ or $\nrule$ \parencite{chellas1980}.
\end{remark}

\section{Supplementation}

\begin{definition}\label{def:supplemented}
  $\fn$ is \textit{supplemented} if $X \in \fn(w)$ and
  $X \subseteq Y$ imply $Y \in \fn(w)$ for all $w$.
\end{definition}

\begin{lemma}\label{lem:supp-closure}
  Every $\fn$ has a unique smallest supplemented extension
  $\fn^+$ given by $\fn^+(w) = \set{Y : X \subseteq Y \text{ for
  some } X \in \fn(w)}$.
\end{lemma}

\begin{proof}
  $\fn(w) \subseteq \fn^+(w)$ is immediate.  If $Y \in
  \fn^+(w)$ and $Y \subseteq Z$, the witnessing $X$ gives
  $X \subseteq Z$, so $Z \in \fn^+(w)$.  Any supplemented
  extension must contain each such $Y$, giving minimality.
\end{proof}
